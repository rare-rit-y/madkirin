bac_all_original_cores_controlled_by_ROOT_or_ally = {
	custom_trigger_tooltip = {
		tooltip = bac_controlls_all_cores_original_ally_tt
		hidden_trigger = {
			all_of_scopes = {
				array = original_cores
				controller = {
					OR = {
						tag = ROOT
						is_in_faction_with = ROOT
					}
				}
			}
		}
	}
}
